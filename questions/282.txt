Подряд без пробелов перечислите номера строк, содержащих 
синтаксические ошибки
<code>
byte b = 0,                             //1
c = b;                                  //2
if (c++ > b--)                          //3
    Console.Write(b = !(bool)c);        //4
else                                    //5
    Console.Write(c);                   //6
</code>
