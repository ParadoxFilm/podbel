В результате выполнения следующего фрагмента программы:
<code>
static int Method(params int[] x) {
    int z = 0;
    foreach (int d in x)
        z += d;
    return z;
}
static void Main() {
    int a = 2;
    Console.Write("{0:d2}:{1:d2}",
        Method(a, ++a, a + 2, 3), Method(3, 2, ++a));
}
</code>
