using System;
class A	{
	int x = 1;
	protected A() { Console.Write("2" + x++); }
	public virtual void gg() { Console.Write("gg" + x); }
}
class B : A	{
	int x = 4;	
	public void gg(int z) { x = 2 * z; }
}
class Test {
	static void Main() {
		B b = new B();
		b.gg(3);
		b.gg();
	}
}
